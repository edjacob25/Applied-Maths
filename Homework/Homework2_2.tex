\section*{2.2.2 Propositional Logic.}

        \centering
        \begin{tabular}{|c|c|c|}
            \hline

            1 & 0 & 1 \\
            \hline
            0 & 1 & 1 \\
            \hline
            0 & 0 & 1 \\
            \hline
        \end{tabular}
    \end{table}
